\section{Solution}
\[ F_a = 15.0N \]
\[ F_b = 12.5N \]
\[ F_c = 16.5N \]
\[ 15\cos(90)+12.5\cos(0)+16.5\cos(225)=.832 \]
\[ 15\sin(90)+12.5\sin(0)+16.5\sin(225)=3.332 \]
\[ \sum F=ma \]
\[ \frac{.832}{.4}=a \]
\[ a=\frac{\triangle{v}}{\triangle{t}} \]
\[ v_{fx}=\frac{.832}{.4}*1 \]
\[ \bar{v_x} = \frac{v_{ix}+v_{fx}}{2} \]
\[ \bar{v_x} = \frac{0+\frac{.832}{.4}*1}{2} \]
\[ \bar{v_x} = 1m/s \]
\[ s_x=\bar{v_x}t \]
\[ s_x=1*1 \]
\[ s_x=1m \]
